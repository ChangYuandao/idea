import numpy as np

def trajectory_evaluate(trajectory_a, trajectory_b):
    """
    Both trajectory_a and trajectory_b are lists, each containing multiple states,
    and each state is represented as a dictionary.

    The label_list is a list of indices that corresponds to the states in the subtrajectory
    where A is definitively better or worse than B.

    The label indicates the quality of the sub-trajectory:
      - if A is better, the label is 1 ("Former")
      - if B is better, the label is -1 ("Latter")
      - if no valid sub-trajectory can be found between the two trajectories, label_list will be an empty list ([]).
    """
    traj_a_length = len(trajectory_a)
    traj_b_length = len(trajectory_b)
    assert traj_a_length == traj_b_length

    label_list = []

    for state_index in range(traj_a_length):
        state_a_info = trajectory_a[state_index]
        state_b_info = trajectory_b[state_index]

        # Extract object and goal positions and rotations
        object_pos_a = state_a_info['object_pos'][:, :3]  # shape: (batch, 3)
        object_rot_a = state_a_info['object_rot'][:, :4]  # shape: (batch, 4)
        goal_pos_a = state_a_info['goal_pos'][:, :3]      # shape: (batch, 3)
        goal_rot_a = state_a_info['goal_rot'][:, :4]      # shape: (batch, 4)
        
        object_pos_b = state_b_info['object_pos'][:, :3]  # shape: (batch, 3)
        object_rot_b = state_b_info['object_rot'][:, :4]  # shape: (batch, 4)
        goal_pos_b = state_b_info['goal_pos'][:, :3]      # shape: (batch, 3)
        goal_rot_b = state_b_info['goal_rot'][:, :4]      # shape: (batch, 4)

        # Calculate orientation distance (angular difference between object and goal rotations)
        # Using quaternion dot product to measure similarity
        dot_product_a = np.sum(object_rot_a * goal_rot_a, axis=1)  # (batch,)
        dot_product_b = np.sum(object_rot_b * goal_rot_b, axis=1)  # (batch,)
        
        # Convert to angular distance (1 - |dot_product| is proportional to angular error)
        # Higher dot product means closer orientation (better)
        orientation_similarity_a = np.abs(dot_product_a)  # Values in [0, 1], higher is better
        orientation_similarity_b = np.abs(dot_product_b)  # Values in [0, 1], higher is better

        # Calculate object position distance to goal (for positional accuracy)
        pos_dist_a = np.linalg.norm(object_pos_a - goal_pos_a, axis=1)  # (batch,)
        pos_dist_b = np.linalg.norm(object_pos_b - goal_pos_b, axis=1)  # (batch,)
        
        # For position distance, lower is better, so we use the inverse (or negative for comparison)
        # Convert to similarity measure where higher is better
        pos_similarity_a = -pos_dist_a  # Higher (less negative) means closer
        pos_similarity_b = -pos_dist_b  # Higher (less negative) means closer

        # For spinning task, the primary metric is orientation similarity
        # Secondary metric is position accuracy
        if orientation_similarity_a >= orientation_similarity_b and pos_similarity_a >= pos_similarity_b:
            label_list.append(1)
        elif orientation_similarity_a <= orientation_similarity_b and pos_similarity_a <= pos_similarity_b:
            label_list.append(-1)
        else:
            label_list.append(0)

    return label_list
