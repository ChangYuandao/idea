import numpy as np

def trajectory_evaluate(trajectory_a, trajectory_b):
    """
    Both trajectory_a and trajectory_b are lists, each containing multiple states,
    and each state is represented as a dictionary.

    The label_list is a list of indices that corresponds to the states in the subtrajectory
    where A is definitively better or worse than B.

    The label indicates the quality of the sub-trajectory:
      - if A is better, the label is 1 ("Former")
      - if B is better, the label is -1 ("Latter")
      - if no valid sub-trajectory can be found between the two trajectories, label_list will be an empty list ([]).
    """
    traj_a_length = len(trajectory_a)
    traj_b_length = len(trajectory_b)
    assert traj_a_length == traj_b_length

    label_list = []

    for state_index in range(traj_a_length):
        state_a_info = trajectory_a[state_index]
        state_b_info = trajectory_b[state_index]

        # Extract object and goal positions
        object_pos_a = state_a_info['object_pos'][:, :3]  # shape: (N, 3)
        object_pos_b = state_b_info['object_pos'][:, :3]  # shape: (N, 3)
        
        goal_pos_a = state_a_info['goal_pos'][:, :3]  # shape: (N, 3)
        goal_pos_b = state_b_info['goal_pos'][:, :3]  # shape: (N, 3)
        
        # Extract object and goal rotations
        object_rot_a = state_a_info['object_rot'][:, :4]  # shape: (N, 4) - quaternion
        object_rot_b = state_b_info['object_rot'][:, :4]  # shape: (N, 4) - quaternion
        
        goal_rot_a = state_a_info['goal_rot'][:, :4]  # shape: (N, 4) - quaternion
        goal_rot_b = state_b_info['goal_rot'][:, :4]  # shape: (N, 4) - quaternion

        # Calculate orientation error (angular distance between object and goal orientation)
        # Using quaternion dot product to measure alignment
        dot_product_a = np.sum(object_rot_a * goal_rot_a, axis=1)  # (N,)
        dot_product_b = np.sum(object_rot_b * goal_rot_b, axis=1)  # (N,)
        
        # Convert to absolute value to handle antipodal quaternions
        dot_product_a = np.abs(dot_product_a)
        dot_product_b = np.abs(dot_product_b)
        
        # Higher dot product means better alignment (lower angular distance)
        # So we want the orientation error metric to be lower for better alignment
        orientation_error_a = 1.0 - dot_product_a  # Lower is better
        orientation_error_b = 1.0 - dot_product_b  # Lower is better

        # Calculate position distance (distance from object to goal position)
        pos_dist_a = np.linalg.norm(object_pos_a - goal_pos_a, axis=1)  # (N,)
        pos_dist_b = np.linalg.norm(object_pos_b - goal_pos_b, axis=1)  # (N,)

        # For goal-oriented evaluation, we primarily care about orientation alignment
        # Position is secondary if the object needs to stay in place
        if orientation_error_a < orientation_error_b and pos_dist_a <= pos_dist_b:
            label_list.append(1)
        elif orientation_error_a > orientation_error_b and pos_dist_a >= pos_dist_b:
            label_list.append(-1)
        else:
            label_list.append(0)

    return label_list
